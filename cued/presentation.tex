	\begin{frame}{Inference procedure}
		We want to draw:
		\begin{equation}
			\theta^{(t)} \sim p(\theta| A, X).
		\end{equation}
		We achieve this by:
		\begin{align}
			b^{(t)} &\sim p \Big( b| A, X \Big) \\
			\theta^{(t)} &\sim p \Big( \theta| X, b^{(t)} \Big)
		\end{align}
	\end{frame}
